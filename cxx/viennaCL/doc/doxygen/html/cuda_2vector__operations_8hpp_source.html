00698       template <typename T>
00699       __global__ void inner_prod_kernel(const T * vec1,
00700                                         unsigned int start1,
00701                                         unsigned int inc1,
00702                                         unsigned int size1,
00703                                         const T * vec2,
00704                                         unsigned int start2,
00705                                         unsigned int inc2,
00706                                         unsigned int size2,
00707                                         T * group_buffer)
00708       {
00709         __shared__ T tmp_buffer[128]; 
00710         unsigned int group_start1 = (blockIdx.x * size1) / (gridDim.x) * inc1 + start1;
00711         unsigned int group_start2 = (blockIdx.x * size2) / (gridDim.x) * inc2 + start2;
00712         
00713         unsigned int group_size1 = ((blockIdx.x + 1) * size1) / (gridDim.x)
00714                                      - (  blockIdx.x * size1) / (gridDim.x);
00715                                      
00716 
00717         T tmp = 0;
00718         for (unsigned int i = threadIdx.x; i < group_size1; i += blockDim.x)
00719           tmp += vec1[i*inc1+group_start1] * vec2[i*inc2+group_start2];
00720         tmp_buffer[threadIdx.x] = tmp;
00721 
00722         // parallel reduction
00723         for (unsigned int stride = blockDim.x/2; stride > 0; stride /= 2)
00724         {
00725           __syncthreads();
00726           if (threadIdx.x < stride)
00727             tmp_buffer[threadIdx.x] += tmp_buffer[threadIdx.x+stride];
00728         }
00729         
00730         if (threadIdx.x == 0)
00731           group_buffer[blockIdx.x] = tmp_buffer[0];
00732         
00733       }

00738       template <typename T>
00739       __global__ void vector_sum_kernel(
00740                 T * vec1,
00741                 unsigned int start1,
00742                 unsigned int inc1,
00743                 unsigned int size1,
00744                 unsigned int option, //0: use fmax, 1: just sum, 2: sum and return sqrt of sum
00745                 T * result) 
00746       { 
00747         __shared__ T tmp_buffer[128]; 
00748         T thread_sum = 0;
00749         for (unsigned int i = threadIdx.x; i<size1; i += blockDim.x)
00750         {
00751           if (option > 0)
00752             thread_sum += vec1[i*inc1+start1];
00753           else
00754             thread_sum = fmax(thread_sum, fabs(vec1[i*inc1+start1]));
00755         }
00756         
00757         tmp_buffer[threadIdx.x] = thread_sum;
00758 
00759         for (unsigned int stride = blockDim.x/2; stride > 0; stride /= 2)
00760         {
00761           if (threadIdx.x < stride)

00763             if (option > 0)
00764               tmp_buffer[threadIdx.x] += tmp_buffer[threadIdx.x + stride];

00766               tmp_buffer[threadIdx.x] = fmax(tmp_buffer[threadIdx.x], tmp_buffer[threadIdx.x + stride]);
00767           }
00768           __syncthreads();
00769         }
00770         
00771         if (threadIdx.x == 0)
00772         {
00773           if (option == 2)
00774             *result = sqrt(tmp_buffer[0]);
00775           else
00776             *result = tmp_buffer[0];
00777         }

00862       template <typename T>
00863       __device__ T impl_norm_kernel( const T * vec,
00864                                      unsigned int start1,
00865                                      unsigned int inc1,
00866                                      unsigned int size1,
00867                                      unsigned int norm_selector,
00868                                      T * tmp_buffer)
00869       {
00870         T tmp = 0;
00871         if (norm_selector == 1) //norm_1
00872         {
00873           for (unsigned int i = threadIdx.x; i < size1; i += blockDim.x)
00874             tmp += fabs(vec[i*inc1 + start1]);
00875         }
00876         else if (norm_selector == 2) //norm_2
00877         {
00878           T vec_entry = 0;
00879           for (unsigned int i = threadIdx.x; i < size1; i += blockDim.x)
00880           {
00881             vec_entry = vec[i*inc1 + start1];
00882             tmp += vec_entry * vec_entry;
00883           }
00884         }
00885         else if (norm_selector == 0) //norm_inf
00886         {
00887           for (unsigned int i = threadIdx.x; i < size1; i += blockDim.x)
00888             tmp = fmax(fabs(vec[i*inc1 + start1]), tmp);
00889         }
00890         
00891         tmp_buffer[threadIdx.x] = tmp;
00892 
00893         if (norm_selector > 0) //parallel reduction for norm_1 or norm_2:
00894         {
00895           for (unsigned int stride = blockDim.x/2; stride > 0; stride /= 2)
00896           {
00897             __syncthreads();
00898             if (threadIdx.x < stride)
00899               tmp_buffer[threadIdx.x] += tmp_buffer[threadIdx.x+stride];
00900           }
00901           return tmp_buffer[0];
00902         }
00903         
00904         //norm_inf:
00905         for (unsigned int stride = blockDim.x/2; stride > 0; stride /= 2)
00906         {
00907           __syncthreads();
00908           if (threadIdx.x < stride)
00909             tmp_buffer[threadIdx.x] = fmax(tmp_buffer[threadIdx.x], tmp_buffer[threadIdx.x+stride]);
00910         }
00911         
00912         return tmp_buffer[0];
00913       };

01231       __global__ void plane_rotation_kernel(
01232                 T * vec1,
01233                 unsigned int start1,
01234                 unsigned int inc1,
01235                 unsigned int size1,
01236                 T * vec2, 
01237                 unsigned int start2,

01242       {
