00048       // Iterate over panels

00050       {
00051         std::size_t row_start = panel_id * max_block_size;
00052         std::size_t current_block_size = std::min<std::size_t>(A.size1() - row_start, max_block_size);

00055         viennacl::range remainder_range(row_start + current_block_size, A.size1());
00056         
00057         //
00058         // Perform LU factorization on panel:
00059         //
00060         
00061         
00062         // Read from matrix to buffer:
00063         viennacl::backend::memory_read(A.handle(), 
00064                                        sizeof(SCALARTYPE) * row_start          * A.internal_size2(),
00065                                        sizeof(SCALARTYPE) * current_block_size * A.internal_size2(),
00066                                        &(temp_buffer[0]));
00067 
00068         // Factorize (kij-version):
00069         for (std::size_t k=0; k < current_block_size - 1; ++k)
00070         {
00071           for (std::size_t i=k+1; i < current_block_size; ++i)
00072           {

00074             
00075             SCALARTYPE l_ik = temp_buffer[row_start + i * A.internal_size2() + k];
00076             
00077             for (std::size_t j = row_start + k + 1; j < A.size1(); ++j)
00078               temp_buffer[i * A.internal_size2() + j] -= l_ik * temp_buffer[k * A.internal_size2() + j];  // l_ik * a_kj
00079           }
00080         }
00081 
00082         // Write back:
00083         viennacl::backend::memory_write(A.handle(), 
00084                                         sizeof(SCALARTYPE) * row_start          * A.internal_size2(),

00091           // Compute L_12 = [ (U_11)^{T}^{-1} A_{21}^T ]^T
