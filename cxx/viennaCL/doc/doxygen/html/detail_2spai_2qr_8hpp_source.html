00196           template<typename MatrixType, typename VectorType, typename ScalarType>
00197           void householder_vector(const MatrixType& A, unsigned int j, VectorType& v, ScalarType& b){
00198               ScalarType sg;
00199               //
00200               dot_prod(A, j+1, sg); 
00201               copy_vector(A, v, j+1);
00202               ScalarType mu;
00203               v(j) = static_cast<ScalarType>(1.0);
00204               if(sg == 0){
00205                   b = 0;
00206               }
00207               else{
00208                   mu = std::sqrt(A(j,j)*A(j, j) + sg);
00209                   if(A(j, j) <= 0){
00210                       v(j) = A(j, j) - mu;
00211                   }else{
00212                       v(j) = -sg/(A(j, j) + mu);
00213                   }
00214                   b = 2*(v(j)*v(j))/(sg + v(j)*v(j));
00215                   v = v/v(j);
00216               }
00217           }
