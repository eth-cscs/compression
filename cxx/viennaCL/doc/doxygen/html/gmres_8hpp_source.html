00115       template <typename VectorType, typename ScalarType>
00116       void gmres_setup_householder_vector(VectorType const & input_vec, VectorType & hh_vec, ScalarType & beta, ScalarType & mu, std::size_t j)
00117       {
00118         ScalarType input_j = input_vec(j);
00119 
00120         // copy entries from input vector to householder vector:
00121         detail::gmres_copy_helper(input_vec, hh_vec, viennacl::traits::size(hh_vec) - (j+1), j+1);
00122         
00123         ScalarType sigma = viennacl::linalg::norm_2(hh_vec);
00124         sigma *= sigma;
00125         
00126         if (sigma == 0)
00127         {
00128           beta = 0;
00129           mu = input_j;
00130         }
00131         else
00132         {
00133           mu = std::sqrt(sigma + input_j*input_j);
00134           
00135           ScalarType hh_vec_0 = (input_j <= 0) ? (input_j - mu) : (-sigma / (input_j + mu));
00136           
00137           beta = ScalarType(2) * hh_vec_0 * hh_vec_0 / (sigma + hh_vec_0 * hh_vec_0);
00138           
00139           //divide hh_vec by its diagonal element hh_vec_0
00140           hh_vec /= hh_vec_0;
00141           hh_vec[j] = ScalarType(1);
00142         }
00143       }

00145       // Apply (I - beta h h^T) to x (Householder reflection with Householder vector h)
00146       template <typename VectorType, typename ScalarType>
00147       void gmres_householder_reflect(VectorType & x, VectorType const & h, ScalarType beta)
00148       {
00149         ScalarType hT_in_x = viennacl::linalg::inner_prod(h, x);

00171       VectorType result(problem_size);
00172       viennacl::traits::clear(result);
00173 
00174       unsigned int krylov_dim = tag.krylov_dim();

00176         krylov_dim = problem_size; //A Krylov space larger than the matrix would lead to seg-faults (mathematically, error is certain to be zero already)
00177

00179       VectorType v_k_tilde(problem_size);
00180       VectorType v_k_tilde_temp(problem_size);
00181       
00182       std::vector< std::vector<CPU_ScalarType> > R(krylov_dim, std::vector<CPU_ScalarType>(tag.krylov_dim()));

00185       std::vector<VectorType>      householder_reflectors(krylov_dim, VectorType(problem_size));
00186       std::vector<CPU_ScalarType>  betas(krylov_dim);
00187 
00188       CPU_ScalarType norm_rhs = viennacl::linalg::norm_2(rhs);
00189       
00190       if (norm_rhs == 0) //solution is zero if RHS norm is zero
00191         return result;
00192       
00193       tag.iters(0);
00194       
00195       for (unsigned int it = 0; it <= tag.max_restarts(); ++it)
00196       {
00197         //
00198         // (Re-)Initialize residual: r = b - A*x (without temporary for the result of A*x)
00199         //
00200         res = rhs;
00201         res -= viennacl::linalg::prod(matrix, result);  //initial guess zero
00202         precond.apply(res);
00203         
00204         CPU_ScalarType rho_0 = viennacl::linalg::norm_2(res); 
00205 
00206         //
00207         // Check for premature convergence
00208         //
00209         if (rho_0 / norm_rhs < tag.tolerance() ) // norm_rhs is known to be nonzero here
00210         {
00211           tag.error(rho_0 / norm_rhs);
00212           return result;
00213         }
00214 
00215         //
00216         // Normalize residual and set 'rho' to 1 as requested in 'A Simpler GMRES' by Walker and Zhou.
00217         //
00218         res /= rho_0;
00219         CPU_ScalarType rho = static_cast<CPU_ScalarType>(1.0);
00220         
00221 
00222         //
00223         // Iterate up until maximal Krylove space dimension is reached:
00224         //
00225         unsigned int k = 0;
00226         for (k = 0; k < krylov_dim; ++k)
00227         {

00229 
00230           // prepare storage:
00231           viennacl::traits::clear(R[k]);
00232           viennacl::traits::clear(householder_reflectors[k]);
00233           
00234           //compute v_k = A * v_{k-1} via Householder matrices
00235           if (k == 0)
00236           {
00237             v_k_tilde = viennacl::linalg::prod(matrix, res);
00238             precond.apply(v_k_tilde);
00239           }
00240           else
00241           {
00242             viennacl::traits::clear(v_k_tilde);
00243             v_k_tilde[k-1] = CPU_ScalarType(1);
00244             
00245             //Householder rotations, part 1: Compute P_1 * P_2 * ... * P_{k-1} * e_{k-1}
00246             for (int i = k-1; i > -1; --i)
00247               detail::gmres_householder_reflect(v_k_tilde, householder_reflectors[i], betas[i]);
00248

00250             precond.apply(v_k_tilde_temp);
00251             v_k_tilde = v_k_tilde_temp;
00252 
00253             //Householder rotations, part 2: Compute P_{k-1} * ... * P_{1} * v_k_tilde
00254             for (unsigned int i = 0; i < k; ++i)
00255               detail::gmres_householder_reflect(v_k_tilde, householder_reflectors[i], betas[i]);
00256           }
00257 
00258           //
00259           // Compute Householder reflection for v_k_tilde such that all entries below k-th entry are zero:
00260           //
00261           CPU_ScalarType rho_k_k = 0;
00262           detail::gmres_setup_householder_vector(v_k_tilde, householder_reflectors[k], betas[k], rho_k_k, k);
00263           
00264           //
00265           // copy first k entries from v_k_tilde to R[k] in order to fill k-th column with result of
00266           // P_k * v_k_tilde = (v[0], ... , v[k-1], norm(v), 0, 0, ...) =: (rho_{1,k}, rho_{2,k}, ..., rho_{k,k}, 0, ..., 0);
00267           //
00268           detail::gmres_copy_helper(v_k_tilde, R[k], k);
00269           R[k][k] = rho_k_k;
00270           
00271           //
00272           // Update residual: r = P_k r
00273           // Set zeta_k = r[k] including machine precision considerations: mathematically we have |r[k]| <= rho
00274           // Set rho *= sin(acos(r[k] / rho))
00275           //
00276           detail::gmres_householder_reflect(res, householder_reflectors[k], betas[k]);
00277           
00278           if (res[k] > rho) //machine precision reached
00279             res[k] = rho;
00280           if (res[k] < -rho) //machine precision reached
00281             res[k] = -rho;
00282           projection_rhs[k] = res[k];
00283           
00284           rho *= std::sin( std::acos(projection_rhs[k] / rho) );
00285           
00286           if (std::fabs(rho * rho_0 / norm_rhs) < tag.tolerance())  // Residual is sufficiently reduced, stop here
00287           {
00288             tag.error( std::fabs(rho*rho_0 / norm_rhs) );
00289             ++k;
00290             break;
00291           }
00292         } // for k
00293 
00294         //
00295         // Triangular solver stage:
00296         //
00297 
00298         for (int i=k-1; i>-1; --i)
00299         {
00300           for (unsigned int j=i+1; j<k; ++j)
00301             projection_rhs[i] -= R[j][i] * projection_rhs[j];     //R is transposed
00302             
00303           projection_rhs[i] /= R[i][i];
00304         }
00305         
00306         //
00307         // Note: 'projection_rhs' now holds the solution (eta_1, ..., eta_k)
00308         //
00309         
00310         res *= projection_rhs[0];
00311         
00312         if (k > 0)
00313         {
00314           for (unsigned int i = 0; i < k-1; ++i)
00315             res[i] += projection_rhs[i+1];
00316         }
00317 
00318         //
00319         // Form z inplace in 'res' by applying P_1 * ... * P_{k}
00320         //
00321         for (int i=k-1; i>=0; --i)
00322           detail::gmres_householder_reflect(res, householder_reflectors[i], betas[i]);
00323 
00324         res *= rho_0;
00325         result += res;  // x += rho_0 * z    in the paper
00326 
00327         //
00328         // Check for convergence:
00329         //
00330         tag.error(std::fabs(rho*rho_0 / norm_rhs));
00331         if ( tag.error() < tag.tolerance() )
00332           return result;
00333       }
00334 
00335       return result;
