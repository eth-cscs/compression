00065         inline std::size_t next_power_2(std::size_t n) {
00066             n = n - 1;
00067 
00068             std::size_t power = 1;
00069 
00070             while(power < sizeof(std::size_t) * 8) {
00071                 n = n | (n >> power);
00072                 power *= 2;
00073             }

00075             return n + 1;
00076         }
