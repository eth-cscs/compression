01166         template <typename InternalType1, typename InternalType2>
01167         class amg_slicing
01168         {
01169             typedef typename InternalType1::value_type SparseMatrixType;
01170             typedef typename InternalType2::value_type PointVectorType;    
01171             
01172           public:
01173             // Data structures on a per-processor basis.
01174             boost::numeric::ublas::vector<InternalType1> A_slice;
01175             boost::numeric::ublas::vector<InternalType2> Pointvector_slice;
01176             // Holds the offsets showing the indices for which a new slice begins.
01177             boost::numeric::ublas::vector<boost::numeric::ublas::vector<unsigned int> > Offset;
01178             
01179             unsigned int threads_;
01180             unsigned int levels_;
01181             
01182             void init(unsigned int levels, unsigned int threads = 0)
01183             {
01184               // Either use the number of threads chosen by the user or the maximum number of threads available on the processor.
01185               if (threads == 0)
01186             #ifdef VIENNACL_WITH_OPENMP
01187                 threads_ = omp_get_num_procs();
01188             #else
01189               threads_ = 1;
01190             #endif   
01191               else 
01192                 threads_ = threads;
01193               
01194               levels_ = levels;
01195               
01196               A_slice.resize(threads_);
01197               Pointvector_slice.resize(threads_);
01198               // Offset has threads_+1 entries to also hold the total size
01199               Offset.resize(threads_+1);
01200               
01201               for (unsigned int i=0; i<threads_; ++i)
01202               {
01203                 A_slice[i].resize(levels_);
01204                 Pointvector_slice[i].resize(levels_);
01205                 // Offset needs one more level for the build-up of the next offset
01206                 Offset[i].resize(levels_+1);
01207               }
01208               Offset[threads_].resize(levels_+1);
01209             } //init()
01210             
01211             // Slice matrix A into as many parts as threads are used.
01212             void slice (unsigned int level, InternalType1 const & A, InternalType2 const & Pointvector)
01213             {
01214               // On the finest level, build a new slicing first.
01215               if (level == 0)
01216                 slice_new (level, A);
01217               
01218               // On coarser levels use the same slicing as on the finest level (Points stay together on the same thread on all levels).
01219               // This is necessary as due to interpolation and galerkin product there only exist connections between points on the same thread on coarser levels.
01220               // Note: Offset is determined in amg_coarse_rs0() after fine level was built.
01221               slice_build (level, A, Pointvector);
01222             }
01223 
01224             // Join point data structure into Pointvector
01225             void join (unsigned int level, InternalType2 & Pointvector) const
01226             {
01227               typedef typename InternalType2::value_type PointVectorType;
01228               
01229               // Reset index offset of all points and update overall C and F point count
01230               Pointvector[level].clear_cf();
01231               for (typename PointVectorType::iterator iter = Pointvector[level].begin(); iter != Pointvector[level].end(); ++iter)
01232               {
01233                 (*iter)->set_offset(0);
01234                 Pointvector[level].update_cf(*iter);
01235               }
01236             }
01237               
01238           private:     
01243             void slice_new (unsigned int level, InternalType1 const & A)
01244             {  
01245               typedef typename SparseMatrixType::const_iterator1 ConstRowIterator;
01246               typedef typename SparseMatrixType::const_iterator2 ConstColIterator;
01247               
01248               // Determine index offset of all the slices (index of A[level] when the respective slice starts).
01249             #ifdef VIENNACL_WITH_OPENMP
01250               #pragma omp parallel for 
01251             #endif
01252               for (unsigned int i=0; i<=threads_; ++i)
01253               {
01254                 // Offset of first piece is zero. Pieces 1,...,threads-1 have equal size while the last one might be greater.
01255                 if (i == 0) Offset[i][level] = 0;
01256                 else if (i == threads_) Offset[i][level] = A[level].size1();
01257                 else Offset[i][level] = i*(A[level].size1()/threads_);
01258               }
01259             }   
01260             
01266             void slice_build (unsigned int level, InternalType1 const & A, InternalType2 const & Pointvector)
01267             {
01268               typedef typename SparseMatrixType::const_iterator1 ConstRowIterator;
01269               typedef typename SparseMatrixType::const_iterator2 ConstColIterator;
01270               
01271               unsigned int x, y;
01272               amg_point *point;
01273               
01274             #ifdef VIENNACL_WITH_OPENMP
01275               #pragma omp parallel for private (x,y,point)
01276             #endif
01277               for (unsigned int i=0; i<threads_; ++i)
01278               {
01279                 // Allocate space for the matrix slice and the pointvector.
01280                 A_slice[i][level] = SparseMatrixType(Offset[i+1][level]-Offset[i][level],Offset[i+1][level]-Offset[i][level]);
01281                 Pointvector_slice[i][level] = PointVectorType(Offset[i+1][level]-Offset[i][level]);
01282                 
01283                 // Iterate over the part that belongs to thread i (from Offset[i][level] to Offset[i+1][level]).
01284                 ConstRowIterator row_iter = A[level].begin1();
01285                 row_iter += Offset[i][level];
01286                 x = row_iter.index1();
01287                     
01288                 while (x < Offset[i+1][level] && row_iter != A[level].end1())
01289                 {
01290                   // Set offset for point index and save point for the respective thread
01291                   point = Pointvector[level][x];
01292                   point->set_offset(Offset[i][level]);
01293                   Pointvector_slice[i][level].add_point(point);
01294                   
01295                   ConstColIterator col_iter = row_iter.begin();
01296                   y = col_iter.index2();
01297                   
01298                   // Save all coefficients from the matrix slice
01299                   while (y < Offset[i+1][level] && col_iter != row_iter.end())
01300                   {
01301                     if (y >= Offset[i][level])
01302                 A_slice[i][level](x-Offset[i][level],y-Offset[i][level]) = *col_iter;
01303                     
01304                     ++col_iter;
01305                     y = col_iter.index2();
01306                   }
01307                   
01308                   ++row_iter;
01309                   x = row_iter.index1();
01310                 }
01311               }
01312             }
01313         };

01320         template <typename SparseMatrixType>
01321         void amg_mat_prod (SparseMatrixType & A, SparseMatrixType & B, SparseMatrixType & RES)
01322         {
01323           typedef typename SparseMatrixType::value_type ScalarType;
01324           typedef typename SparseMatrixType::iterator1 InternalRowIterator;
01325           typedef typename SparseMatrixType::iterator2 InternalColIterator;
01326           
01327           unsigned int x,y,z;
01328           ScalarType prod;
01329           RES = SparseMatrixType(A.size1(), B.size2());
01330           RES.clear();
01331           
01332     #ifdef VIENNACL_WITH_OPENMP
01333           #pragma omp parallel for private (x,y,z,prod) shared (A,B,RES)
01334     #endif
01335           for (x=0; x<A.size1(); ++x)
01336           {
01337             InternalRowIterator row_iter = A.begin1();
01338             row_iter += x;
01339             for (InternalColIterator col_iter = row_iter.begin(); col_iter != row_iter.end(); ++col_iter)
01340             {
01341               y = col_iter.index2(); 
01342               InternalRowIterator row_iter2 = B.begin1();
01343               row_iter2 += y;
01344 
01345               for(InternalColIterator col_iter2 = row_iter2.begin(); col_iter2 != row_iter2.end(); ++col_iter2)
01346               {
01347                 z = col_iter2.index2();
01348                 prod = *col_iter * *col_iter2;
01349                 RES.add(x,z,prod);
01350               }
01351             }
01352           }
01353         }
