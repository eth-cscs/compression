00702         template<typename ScalarType, unsigned int MAT_ALIGNMENT>
00703         void row_info(coordinate_matrix<ScalarType, MAT_ALIGNMENT> const & mat,
00704                       vector_base<ScalarType> & vec,
00705                       viennacl::linalg::detail::row_info_types info_selector)
00706         {
00707           ScalarType         * result_buf   = detail::extract_raw_pointer<ScalarType>(vec.handle());
00708           ScalarType   const * elements     = detail::extract_raw_pointer<ScalarType>(mat.handle());
00709           unsigned int const * coord_buffer = detail::extract_raw_pointer<unsigned int>(mat.handle12());
00710           
00711           ScalarType value = 0;
00712           unsigned int last_row = 0;
00713           
00714           for (std::size_t i = 0; i < mat.nnz(); ++i)
00715           {
00716             unsigned int current_row = coord_buffer[2*i];
00717             
00718             if (current_row != last_row)
00719             {
00720               if (info_selector == viennacl::linalg::detail::SPARSE_ROW_NORM_2)
00721                 value = std::sqrt(value);
00722               
00723               result_buf[last_row] = value;
00724               value = 0;
00725               last_row = current_row;
00726             }
00727             
00728             switch (info_selector)
00729             {
00730               case viennacl::linalg::detail::SPARSE_ROW_NORM_INF: //inf-norm
00731                 value = std::max<ScalarType>(value, std::fabs(elements[i]));
00732                 break;
00733                 
00734               case viennacl::linalg::detail::SPARSE_ROW_NORM_1: //1-norm
00735                 value += std::fabs(elements[i]);
00736                 break;
00737                 
00738               case viennacl::linalg::detail::SPARSE_ROW_NORM_2: //2-norm
00739                 value += elements[i] * elements[i];
00740                 break;
00741                 
00742               case viennacl::linalg::detail::SPARSE_ROW_DIAGONAL: //diagonal entry
00743                 if (coord_buffer[2*i+1] == current_row)
00744                   value = elements[i];
00745                 break;
00746                 
00747               default:
00748                 break;
00749             }
00750           }
00751           
00752           if (info_selector == viennacl::linalg::detail::SPARSE_ROW_NORM_2)
00753             value = std::sqrt(value);
00754           
00755           result_buf[last_row] = value;
00756         }
