00044     // function to calculate the increment of combination comb.
00045     // parameters:
00046     // comb: pointer to vector<int> of size m, m <= n
00047     //       1 <= comb[i] <= n for 0 <= i < m
00048     //       comb[i] < comb[i+1] for 0 <= i < m - 1
00049     //       comb represents an unordered selection of m values out of n
00050     // n: int
00051     //    total number of values out of which comb is taken as selection
00052     inline bool comb_inc(std::vector<int> & comb, int n)
00053     {
00054         int m;
00055         int k;
00056         
00057         m = comb.size();
00058         // calculate k as highest possible index such that (*comb)[k-1] can be incremented
00059         k = m;
00060         while ( (k > 0) && ( ((k == m) && (comb[k-1] == n)) || 
00061                             ((k < m) && (comb[k-1] == comb[k] - 1) )) )
00062         {
00063             k--;
00064         }
00065         if (k == 0) // no further increment of comb possible -> return false
00066         {
00067             return false;
00068         }
00069         else
00070         {
00071             (comb[k-1])++; // increment (*comb)[k-1],
00072             for (int i = k; i < m; i++) // and all higher index positions of comb are 
00073             // calculated just as directly following integer values (lowest possible values)
00074             {
00075                 comb[i] = comb[k-1] + (i - k + 1);
00076             }
00077             return true;
00078         }
00079     }
